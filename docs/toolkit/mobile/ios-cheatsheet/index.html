Get iDevice UDID

# option 1
ioreg -p IOUSB -l | grep "USB Serial" 

# option 2
idevice_id -l # get UDID with ideviceinstaller

# option 3
system_profiler SPUSBDataType | sed -n -e '/iPad/,/Serial/p;/iPhone/,/Serial/p;/iPod/,/Serial/p' | grep "Serial Number:" # 

# option 4
instruments -s devices
